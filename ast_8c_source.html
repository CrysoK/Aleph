57void ast_free(AstNode *node) {
58 print_dbg("(%s) liberando %s", __func__, node ? ast_enum_str(node->type) : "NULL");
59 if(!node) return;
60 AstDataN *data = (AstDataN *)node;
61 AstExprN *expr = (AstExprN *)node;
62 AstStmtN *stmt = (AstStmtN *)node;
63 AstIdN *id = (AstIdN *)node;
64 AstIfnN *ifn = (AstIfnN *)node;
65 switch(node->type) {
66 case STR_AST:
67 free(data->str); data->str = NULL;
68 // fallthrough
69 case INT_AST:
70 free(data);
71 break;
72 case SET_AST: case LST_AST:
73 ast_free(data->elems); data->elems = NULL;
74 free(data);
75 break;
76 case OR_AST: case AND_AST: case EQUAL_AST: case LESS_AST: case IN_AST:
77 case ELEM_AST: case ADD_AST: case SUB_AST: case MUL_AST: case DIV_AST:
78 case CALL_AST:
79 ast_free(expr->a); expr->a = NULL;
80 ast_free(expr->b); expr->b = NULL;
81 free(expr);
82 break;
83 case NOT_AST: case NEG_AST: case LEN_AST:
84 ast_free(expr->x); expr->x = NULL;
85 free(expr);
86 break;
87 case INSERT_AST: case IF_AST: case FOR_AST: case FN_AST:
88 ast_free(stmt->c); stmt->c = NULL;
89 // fallthrough
90 case DECL_AST: case ASSIGN_AST: case WHILE_AST:
91 ast_free(stmt->b); stmt->b = NULL;
92 // fallthrough
93 case EXPR_AST: case RETURN_AST: case DEL_AST:
94 ast_free(stmt->a); stmt->a = NULL;
95 // fallthrough
96 case BREAK_AST: case CONTINUE_AST: case EXIT_AST:
97 free(stmt);
98 break;
99 case ID_AST: free(id->str); free(id); break;
100 case IFN_AST: free(ifn); break;
101 case AST_LL:
102 ast_free(node->data); node->data = NULL;
103 ast_free(node->next); node->next = NULL;
104 free(node);
105 break;
106 default:
107 print_error(
108 NULL, "Interno", "No se pudo liberar el tipo %s", ast_enum_str(node->type));
109 }
110}
