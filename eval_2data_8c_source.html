295int data_compare(ResNode *res_a, ResNode *res_b) {
296 int r = 0;
297 // Chequeo de nulos
298 if(!res_a || !res_b) {
299 print_error("Interno", "(%s) Algún parámetro es NULL", __func__);
300 goto end;
301 }
302 if(!res_is_data(res_a) || !res_is_data(res_b)) {
303 print_error("Interno", "(%s) Algún parámetro es inválido", __func__);
304 goto end;
305 }
306 // ¿Son el mismo dato?
307 if(res_a == res_b) { r = 0; goto end; }
308 // Orden si el tipo es diferente
309 r = (int)res_a->type - (int)res_b->type;
310 if(r != 0) goto end;
311 // Orden si son del mismo tipo
312 switch(res_a->type) {
313 case LST_R: case SET_R: // Si son iterables
314 // Orden si el tamaño es diferente
315 r = data_len(res_a) - data_len(res_b);
316 if(r != 0) break;
317 // Orden si el tamaño es igual (comparar lista de elementos)
318 r = compare_data_list(data_elems(res_a), data_elems(res_b));
319 break;
320 case INT_R: // Si son elementos (enteros)
321 r = data_int(res_a) - data_int(res_b);
322 break;
323 case STR_R: // Si son elementos (cadenas)
324 r = strcmp(data_str(res_a), data_str(res_b));
325 break;
326 case VOID_R: case UNDEF_R: // Si es Void o Undef
327 r = 0; // Void == Void e Undef == Undef
328 break;
329 default:
330 // En este punto, el tipo es inválido
331 print_error("Interno", "(%s) Nodo inválido", __func__);
332 }
333
334 end:
335 // Se usa una variable de retorno para poder mostrarla al final
336 if(g_debug) {
337 printf("DEBUG: (%s) ", __func__);
338 data_print(res_a, true);
339 if(r == 0) printf(" == ");
340 else if(r < 0) printf(" < ");
341 else printf(" > ");
342 data_print(res_b, true);
343 printf("\n");
344 fflush(stdout);
345 }
346 return r;
347}

380bool set_insert_elem(ResNode *data, ResNode **elems) {
381 DBG_FN_START;
382 if(!data) return false;
383 ResNode *new = res_new_ll_n(data);
384 int cmp;
385 // Si el primer elemento es nulo o es mayor al nuevo, insertar en primera
386 // posición
387 if(!(*elems) || (cmp = data_compare((*elems)->data, data)) > 0) {
388 new->next = *elems;
389 *elems = new;
390 return true;
391 }
392 if(!cmp) goto free;
393 // Buscar el nodo anterior a la posición de inserción
394 ResNode *current = *elems;
395 while(current->next && (cmp = data_compare(current->next->data, data)) < 0) {
396 current = current->next;
397 }
398 if(!cmp) goto free;
399 new->next = current->next;
400 current->next = new;
401 return true;
402
403 free:
404 print_dbg("(%s) Elemento repetido", __func__);
405 new->data = NULL;
406 res_free(new);
407 return false;
