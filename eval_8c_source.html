70ResNode *eval_expr_list(AstNode *ast_ll, int *counter) {
71 DBG_FN_START;
72 if(!ast_ll) { return NULL; } // No es un error
73 ResNode *aux = eval(ast_ll->data);
74 NULL_RETURN(aux, "Interno?", S("aux == NULL"));
75 ERROR_RETURN(aux);
76 ResNode *rl = res_new_ll_n(aux);
77 if(counter) (*counter) += 1;
78 ResNode *last = rl;
79 while((ast_ll = ast_ll->next)) {
80 aux = eval(ast_ll->data);
81 NULL_GOTO(aux, aux, "Interno?", S("aux == NULL"), discard);
82 ERROR_GOTO(aux, aux, discard);
83 last->next = res_new_ll_n(aux);
84 last = last->next;
85 if(counter) (*counter) += 1;
86 }
87 return rl;
88
89 discard:
90 res_free(rl);
91 return MAYBE_ERROR(aux);
92}

124void garbage_collect(ResNode *res) {
125 if(!res) return;
126 if(res->type < UNDEF_R || res->type > FN_R) return;
127 if(data_refcnt(res) > 0) return;
128 if(g_debug) { printf("DEBUG GC: "); data_print(res, true); printf("\n"); }
129 ResDataN *data = (ResDataN *)res;
130 ResNode *elem = NULL;
131 switch(res->type) {
132 case UNDEF_R: case VOID_R:
133 free(data); break;
134 case INT_R:
135 free(data);
136 break;
137 case STR_R:
138 free(data->str);
139 data->str = NULL;
140 free(data);
141 break;
142 case SET_R: case LST_R:
143 elem = data->elems;
144 while(elem) {
145 ResNode *next = elem->next;
146 // REF_DEC(elem->data);
147 garbage_collect(elem->data);
148 elem->data = NULL;
149 elem->next = NULL;
150 free(elem);
151 elem = next;
152 }
153 free(data);
154 break;
155 case FN_R:
156 // ast_free se encarga de liberar data->fn al final de la ejecución
157 free(data);
158 break;
159 default:
160 print_error("Interno", "Inalcanzable (tipo %s)", res_enum_str(res->type));
161 }

190void res_free(ResNode *node) {
191 print_dbg("(%s) %s", __func__, node ? res_enum_str(node->type) : "NULL");
192 if(!node) return;
193 switch(node->type) {
194 case UNDEF_R: case VOID_R: case INT_R: case STR_R: case SET_R: case LST_R: case FN_R:
195 garbage_collect(node); // depende de su refcnt
196 break;
197 case RETURN_R: garbage_collect(node->data); free(node); node = NULL; break;
198 case BREAK_R: case CONTINUE_R: case EXIT_R: free(node); break;
199 case ERROR_R: res_free_error(node); break;
200 case RES_LL: res_free_ll(node); break;
201 }
202}
