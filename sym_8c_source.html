173int get_sym_pos(SymTab *symtab, char *id, int hash) {
174 if(!id) {
175 print_error("Interno", LOC_S"'id' = NULL", LOC);
176 return -1;
177 }
178 if(!symtab) {
179 print_error("Fatal", LOC_S"Tabla de símbolos nula", LOC);
180 abort();
181 }
182 // Posición inicial según hashing
183 int init = hash % g_sym_tab_size;
184 int pos = init;
185 // Puntero de la posición inicial
186 Symbol *elem = (symtab->hash_tab)[init];
187 // Si no es NULL, ver si el ID es igual
188 while(elem) {
189 // Si el ID coincide, símbolo encontrado
190 if(strcmp(elem->id, id) == 0) {
191 print_dbg("(%s) pos = %d", __func__, pos);
192 return pos;
193 }
194 print_dbg(LOC_S"Colisión en el hashing", LOC);
195 // Si el ID es distinto, buscar en la siguiente posición
196 pos = (pos + 1) % g_sym_tab_size;
197 if(pos != init) elem = (symtab->hash_tab)[pos];
198 // Colisión en toda la tabla (¿posible?)
199 else break;
200 }
201 // Si es NULL, el símbolo nunca se agregó en la tabla actual
202 return -1;
203}

206Symbol *scope_get_sym(char *id, int hash) {
207 print_dbg("(%s) id = %s", __func__, id ? id : "NULL");
208 if(!id) return NULL;
209 if(!g_symtab) { // Podría ocurrir?
210 res_free(RAISE("Fatal", S("Tabla de símbolos nula")));
211 abort();
212 }
213 SymTab *s_tab = g_symtab;
214 int pos = get_sym_pos(s_tab, id, hash);
215 while(s_tab->prev && pos < 0) { // Buscar en alcances anteriores
216 s_tab = s_tab->prev;
217 pos = get_sym_pos(s_tab, id, hash);
218 }
219 if(pos >= 0) {
220 return (s_tab->hash_tab)[pos];
221 }
222 return NULL;
223}

235ResNode *scope_add_sym(char *id, int hash, ResNode * value) {
236 DBG_FN_START;
237 NULL_RETURN(id, "Interno", S("id == NULL"));
238 NULL_RETURN(value, "Interno", S("value == NULL"));
239 if(!g_symtab) { // "fatal" o no?
240 res_free(RAISE("Fatal", S("Tabla de símbolos nula")));
241 abort();
242 }
243 // Posición inicial según hashing
244 int init = hash % g_sym_tab_size;
245 int pos = init;
246 // Puntero de la posición inicial
247 Symbol *elem = g_symtab->hash_tab[init];
248 // Hasta que no se encuentre una posición válida, repetir
249 while(true) {
250 // Si es NULL, es una posición válida
251 if(!elem) {
252 g_symtab->hash_tab[pos] = res_new_sym(id, value);
253 REF_INC(value);
254 id_list_add(id, hash);
255 print_dbg("(%s) Símbolo '%s' añadido", __func__, id);
256 return NULL;
257 } // Sino, ya hay algo con el mismo hash
258 // Si el ID coincide es una re-declaración
259 if(strcmp(elem->id, id) == 0) { // es error de sintaxis?
260 return RAISE("Sintaxis", str_fmt("Re-declaración del símbolo '%s'", id));
261 } // Si los ID no coinciden, hay una colisión. Buscar sig. pos. válida
262 print_dbg("Colisión en el hashing");
263 pos = (pos + 1) % g_sym_tab_size;
264 if(pos != init) elem = g_symtab->hash_tab[pos];
265 else { // Si pos == init, entonces se recorrió toda la tabla
266 return RAISE("Interno", S("Tabla de símbolos llena"));
267 }
268 }
