329ResNode *eval_fncall_params(
330 ResNode *args, AstNode *fndef_ast, int a_cnt, char *fname
331) {
332 DBG_FN_START;
333 int p_cnt = 0; // cantidad de parámetros
334 bool varargs = false; // cantidad de parámetros indeterminada
335 ERROR_RETURN(args); // puede ser NULL
336 ResNode *arg = args;
337 AstNode *param = ast_stmt_b(fndef_ast);
338 ResNode *r = NULL;
339 while(param) {
340 char *id = ast_id_str(param->data);
341 unsigned hash = ast_id_hash(param->data);
342 if(strcmp(id, "...") == 0) { varargs = true; break; }
343 if(arg) { // permite seguir contando parámetros aunque no queden argumentos
344 if(!arg->data) // args puede ser nulo (llamada sin argumentos) pero UN arg no.
345 return RAISE("Valor", S("Argumento inválido"));
346 r = scope_add_sym(id, hash, arg->data);
347 if(res_is_error(r)) return MAYBE_ERROR(r);
348 arg = arg->next;
349 }
350 param = param->next; p_cnt++;
351 }
352 if(varargs && p_cnt <= a_cnt) { // Guarda todos los argumentos restantes en `_args`
353 // <= porque debe poder se una lista vacía (ej. print())
354 ResNode * _args = res_new_lst(arg, a_cnt - p_cnt);
355 r = scope_add_sym("_args", str_hash("_args"), _args);
356 if(res_is_error(r)) {
357 ((ResDataN *)_args)->elems = NULL; // deja el trabajo para free_args
358 res_free(_args);
359 return MAYBE_ERROR(r);
360 }
361 p_cnt = a_cnt;
362 }
363 if(p_cnt != a_cnt) {
364 r = RAISE("Tipo", str_fmt(
365 "'%s' necesita %s %d argumento%s (%d dado%s)", fname,
366 varargs ? "al menos" : "exactamente", p_cnt,
367 p_cnt > 1 ? "s" : "", a_cnt, a_cnt > 1 ? "s" : "")
368 );
369 }
370 return r;
371}

374 DBG_FN_START;
375 NULL_RETURN(fncall_ast, "Interno", S("fncall_ast == NULL"));
376 ResNode *r = NULL;

382 goto free_sym_val;
383 }

385 NULL_GOTO(fndef_ast, r, "Interno", S("fndef_ast == NULL"), free_sym_val);
386 int a_cnt = 0; // cantidad de argumentos
387 ResNode *args = eval_expr_list(ast_stmt_b(fncall_ast), &a_cnt); // liberar
388 ERROR_GOTO(args, r, free_sym_val);
389 int calling_depth = scope_depth(); // Establece el scope global como actual
390 SymTab *calling_scope = scope_change(NULL, 0);
391 scope_push(); // Scope local de la función
392 r = eval_fncall_params(
393 args, fndef_ast, a_cnt, ast_id_str(ast_expr_a(fncall_ast)));
394 ERROR_GOTO(r, r, end_scope);
395 // Evaluación de sentencias
396 r = eval(ast_stmt_c(fndef_ast));
397 print_dbg("fncall = %s", r ? res_enum_str(r->type) : "NULL");
398 ERROR_GOTO(r, r, end_scope); // Incluye `exit`
399 scope_pop(); // Fin scope local.
400 scope_change(calling_scope, calling_depth); // Restablece el scope previo
401 ResNode *aux = r;
402 if(r && r->type == RETURN_R && r->data) {
403 r = r->data; // return con algún dato
404 REF_DEC(r); // el correspondiente +1 está en eval_stmt
405 aux->data = NULL;// deja de referenciar el dato, para que res_free no lo libere
406 } else r = res_new_void();
407 res_free(aux);
408 return r;
409
410 end_scope:
411 res_free(args);
412 scope_pop(); // Fin scope local
413 scope_change(calling_scope, calling_depth); // Restablece el scope previo
414 free_sym_val:
415 res_free(sym_val);
416 return r;

607// Como los conjuntos están ordenados, se puede decir que un elemento de 'a' no
608// está en 'b' si es menor al que se comparó
609// - {(1), 3, 4} * {(2), 3, 5} = {}
610// - 1 no puede estar en 'b' porque 2 es mayor
611// - {1, (3), 4} * {(2), 3, 5} = {}
612// - 2 no puede estar en 'a' porque 3 es mayor
613// - {1, (3), 4} * {2, (3), 5} = {3}
614// - {1, 3, (4)} * {2, 3, (5)} = {3}
615// - 4 no puede estar en 'b' porque 5 es mayor
616// - {1, 3, 4} * {2, 3, 5} = {3}
617ResNode *set_new_inters(ResNode *a, ResNode *b, int *new_len) {
618 DBG_FN_START;
